//计算n的阶乘, int a[]保存结果, 返回阶乘位数digits

#include <memory.h>
int a[10000];   //确保保存最终运算结果的数组足够大
int bignumber_factorial(int n, int *a)
{
    memset(a, 0, sizeof(a)), a[0] = 1; 
    int nCarry, nDigits = 1; //进位, 位数
    int temp;     
    for(int i = 2; i <= n; ++i) {
        //将临时结果的每位与阶乘元素相乘 
        for(int j = 1, nCarry = 0; j <= nDigits; ++j) {
            temp = a[j-1] * i + nCarry;
            a[j-1] = temp % 10; 
            nCarry = temp / 10; 
        }
        //进位
        while(nCarry) {
            a[++nDigits-1] = nCarry % 10;
            nCarry /= 10;
        }
    }
    return nDigits;
}
