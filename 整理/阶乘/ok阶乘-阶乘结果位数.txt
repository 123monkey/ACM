#include <iostream>
#include <cmath>
using namespace std;

//阶乘位数-stirling公式
//返回n!的十进制位数

int factorial_digits(int n)
{
    static const double e = 2.7182818284590452354, pi = 3.141592653589793239;
    double Digits = 0.0;
    int ret;
    
    if (n < 100000) {
        for (int i = 1; i <= n; ++i)
            Digits += log10(i*1.0) ;
    }        
    else {        
        //1.整数位数求法(int)long10(n)+1
        //2.stirling公式---因为是一个近似公式,所以
        //  当输入m较小的时候,不能够使用公式直接计算
        // 把公式中的sqrt漏掉了,害我改了好久!!!!!!!
        Digits = log10(sqrt(2*pi*n)) + n*log10(n/e);
    }
    ret = int(Digits);
    if(ret <= Digits) ++ret;
    return ret;
}
